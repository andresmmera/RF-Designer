112struct TraceInfo {
113 QString dataset;
114 QString parameter;
115 DisplayMode displayMode;
116
119 QString uniqueId() const {
120 if ((displayMode == DisplayMode::Magnitude_dB) ||
121 (displayMode == DisplayMode::Phase)) {
122 QString displayMode_str = QString::number(static_cast<int>(displayMode)); // Assign to variable to avoid warning
123 return QString("%1.%2_%3")
124 .arg(dataset, parameter, displayMode_str);
125 } else {
126 return QString("%1.%2").arg(dataset, parameter);
127 }
128 }
129
132 QString displayName() const {
133 QString modeName = QString("");
134
135 if (displayMode == DisplayMode::Magnitude_dB) {
136 modeName = "dB";
137 } else {
138 if (displayMode == DisplayMode::Phase) {
139 modeName = "Phase";
140 } else {
141 return QString("%1.%2").arg(dataset, parameter);
142 }
143 }
144 return QString("%1.%2_%3").arg(dataset, parameter, modeName);
145 }
146};
